-- Create Books table
CREATE TABLE Books (
    AccNo NUMBER PRIMARY KEY,
    Title VARCHAR2(100),
    Author VARCHAR2(50),
    Publisher VARCHAR2(50),
    Count NUMBER
);

-- Create Library_Audit table
CREATE TABLE Library_Audit (
    AccNo NUMBER,
    Title VARCHAR2(100),
    Author VARCHAR2(50),
    Publisher VARCHAR2(50),
    Count NUMBER,
    Action_Type VARCHAR2(20),
    Action_Date DATE
);
